  <action>
    Create `Math Quiz/Services/QuestionGenerator.swift`:

    ```swift
    import Foundation

    struct QuestionGenerator {
        func generate(config: GameConfig) -> MathQuestion {
            guard !config.selectedOperations.isEmpty else {
                // Fallback: addition with default range
                return makeQuestion(operation: .addition, range: config.operandRange)
            }
            let operation = config.selectedOperations.randomElement()!
            return makeQuestion(operation: operation, range: config.operandRange)
        }

        private func makeQuestion(operation: MathOperation,
                                   range: ClosedRange<Int>) -> MathQuestion {
            switch operation {
            case .addition:
                let a = Int.random(in: range)
                let b = Int.random(in: range)
                return MathQuestion(operand1: a, operand2: b,
                                    operation: .addition, correctAnswer: a + b)

            case .subtraction:
                let a = Int.random(in: range)
                let b = Int.random(in: range)
                // Ensure non-negative result for kids
                let (bigger, smaller) = a >= b ? (a, b) : (b, a)
                return MathQuestion(operand1: bigger, operand2: smaller,
                                    operation: .subtraction, correctAnswer: bigger - smaller)

            case .multiplication:
                let a = Int.random(in: range)
                let b = Int.random(in: range)
                return MathQuestion(operand1: a, operand2: b,
                                    operation: .multiplication, correctAnswer: a * b)

            case .division:
                // Generate divisor and quotient first; compute dividend.
                // Clamp divisor range to 1...max to prevent division by zero.
                let divisorRange = max(1, range.lowerBound)...range.upperBound
                let divisor  = Int.random(in: divisorRange)
                let quotient = Int.random(in: range)
                let dividend = divisor * quotient
                return MathQuestion(operand1: dividend, operand2: divisor,
                                    operation: .division, correctAnswer: quotient)
            }
        }
    }
    ```
