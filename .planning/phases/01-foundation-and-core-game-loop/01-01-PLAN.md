SchemaV1.swift structure:
```swift
import SwiftData

enum SchemaV1: VersionedSchema {
    static var versionIdentifier = Schema.Version(1, 0, 0)
    static var models: [any PersistentModel.Type] { [GameSession.self, QuestionResult.self] }

    @Model final class GameSession {
        var date: Date = Date.now
        var score: Int = 0
    }

    @Model final class QuestionResult {
        var session: GameSession?
        var wasCorrect: Bool = false
    }
}

typealias GameSession = SchemaV1.GameSession
typealias QuestionResult = SchemaV1.QuestionResult
```
