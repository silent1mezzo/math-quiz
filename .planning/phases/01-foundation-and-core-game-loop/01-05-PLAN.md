    Create `Math Quiz/Views/NumberPadView.swift` following Pattern 4 from the research document exactly.

    Key constraints:
    - 12 buttons in this exact order: ["1","2","3","4","5","6","7","8","9","⌫","0","→"]
    - LazyVGrid with 3 columns using GridItem(.flexible(), spacing: 12)
    - Each button: `.frame(maxWidth: .infinity)` + `.frame(height: 70)` for GAME-04 (70pt minimum)
    - "⌫" calls `onDelete()`, "→" calls `onSubmit()`, all digits call `onDigit(label)`
    - No state, no @Observable, no binding to GameViewModel directly — closures only
    - No TextField, no system keyboard, no .focusable that triggers the system keyboard
    - Accessibility: set `.accessibilityLabel` on ⌫ to "Delete" and → to "Submit"

    ```swift
    import SwiftUI

    struct NumberPadView: View {
        let onDigit: (String) -> Void
        let onDelete: () -> Void
        let onSubmit: () -> Void

        private let digits = ["1","2","3","4","5","6","7","8","9","⌫","0","→"]
        private let columns = Array(repeating: GridItem(.flexible(), spacing: 12), count: 3)

        var body: some View {
            LazyVGrid(columns: columns, spacing: 12) {
                ForEach(digits, id: \.self) { label in
                    Button {
                        switch label {
                        case "⌫": onDelete()
                        case "→": onSubmit()
                        default:  onDigit(label)
                        }
                    } label: {
                        Text(label)
                            .font(.title)
                            .frame(maxWidth: .infinity)
                            .frame(height: 70)
                            .background(.tint.opacity(0.15))
                            .clipShape(RoundedRectangle(cornerRadius: 12))
                    }
                    .accessibilityLabel(label == "⌫" ? "Delete" : label == "→" ? "Submit" : label)
                }
            }
            .padding(.horizontal)
        }
    }
    ```

    Create `Math Quiz/Views/QuestionView.swift` following Pattern 7 from the research document:

    ```swift
    import SwiftUI
    import Pow

    struct QuestionView: View {
        let question: MathQuestion
        let answerBuffer: String
        let feedbackState: FeedbackState
        let correctCount: Int
        let incorrectCount: Int

        var body: some View {
            VStack(spacing: 16) {
                Text(question.displayString)
                    .font(.system(size: 48, weight: .bold))
                    .foregroundStyle(foregroundColor)
                    .changeEffect(
                        .spray(origin: .center) {
                            Image(systemName: "star.fill")
                                .foregroundStyle(.yellow)
                        },
                        value: correctCount
                    )
                    .changeEffect(.shake(rate: .fast), value: incorrectCount)

                // Answer buffer display
                Text(answerBuffer.isEmpty ? " " : answerBuffer)
                    .font(.system(size: 36, weight: .medium))
                    .foregroundStyle(.secondary)
                    .frame(minWidth: 80)
                    .padding(.horizontal, 12)
                    .padding(.vertical, 8)
                    .background(.quaternary)
                    .clipShape(RoundedRectangle(cornerRadius: 8))
            }
        }

        private var foregroundColor: Color {
            switch feedbackState {
            case .idle:      return .primary
            case .correct:   return .green
            case .incorrect: return .red
            }
        }
    }
    ```

    The color change on feedbackState (idle → correct/incorrect → idle) provides the visual component of GAME-05. Do NOT wrap in withAnimation — fixed 600ms flash only (anti-pattern from research).

    Create `Math Quiz/Views/ScoreHeaderView.swift`:

    ```swift
    import SwiftUI

    struct ScoreHeaderView: View {
        let elapsedSeconds: Int
        let score: Int

        var body: some View {
            HStack {
                Label(formattedElapsed, systemImage: "clock")
                    .font(.headline)
                    .foregroundStyle(.secondary)

                Spacer()

                Label("\(score)", systemImage: "star.fill")
                    .font(.headline.weight(.bold))
                    .foregroundStyle(.yellow)
            }
            .padding(.horizontal)
        }

        private var formattedElapsed: String {
            let minutes = elapsedSeconds / 60
            let seconds = elapsedSeconds % 60
            return String(format: "%d:%02d", minutes, seconds)
        }
    }
    ```
