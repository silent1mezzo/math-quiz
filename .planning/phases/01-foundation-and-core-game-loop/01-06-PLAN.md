    Create `Math Quiz/Views/GameView.swift`:

    ```swift
    import SwiftUI

    struct GameView: View {
        @Bindable var viewModel: GameViewModel
        @Environment(\.scenePhase) private var scenePhase

        private let timer = Timer.publish(every: 1, on: .main, in: .common).autoconnect()

        // Track answer counts to drive Pow animations
        @State private var correctCount = 0
        @State private var incorrectCount = 0

        var body: some View {
            VStack(spacing: 24) {
                ScoreHeaderView(
                    elapsedSeconds: viewModel.elapsedSeconds,
                    score: viewModel.score
                )

                Spacer()

                if let question = viewModel.currentQuestion {
                    QuestionView(
                        question: question,
                        answerBuffer: viewModel.answerBuffer,
                        feedbackState: viewModel.feedbackState,
                        correctCount: correctCount,
                        incorrectCount: incorrectCount
                    )
                }

                Spacer()

                NumberPadView(
                    onDigit: { viewModel.appendDigit($0) },
                    onDelete: { viewModel.deleteLastDigit() },
                    onSubmit: {
                        let wasCorrect = viewModel.feedbackState == .idle &&
                            viewModel.currentQuestion != nil &&
                            (Int(viewModel.answerBuffer) == viewModel.currentQuestion?.correctAnswer)
                        viewModel.submitAnswer()
                        if wasCorrect { correctCount += 1 } else { incorrectCount += 1 }
                    }
                )
                .padding(.bottom, 16)
            }
            .navigationTitle("Math Quiz")
            .navigationBarTitleDisplayMode(.inline)
            .onReceive(timer) { _ in
                guard viewModel.sessionActive else { return }
                viewModel.tick()
            }
            .onChange(of: scenePhase) { _, newPhase in
                switch newPhase {
                case .background: viewModel.pauseSession()
                case .active:     viewModel.resumeSession()
                default:          break
                }
            }
            .sensoryFeedback(.success, trigger: correctCount)
            .sensoryFeedback(.error,   trigger: incorrectCount)
        }
    }
    ```

    Note on correctCount logic: The animation count increment must happen AFTER submitAnswer() is called so feedbackState has been read. Actually, the simplest approach is to check the answer buffer against currentQuestion before calling submitAnswer, then call submitAnswer. The exact implementation detail is up to the executor as long as: (a) correctCount increments on correct submission, (b) incorrectCount increments on incorrect submission.

    Create `Math Quiz/Views/OperationPickerView.swift`:

    ```swift
    import SwiftUI

    struct OperationPickerView: View {
        @State private var config = GameConfig.default
        var onStart: (GameConfig) -> Void

        var body: some View {
            VStack(spacing: 32) {
                Text("Choose Operations")
                    .font(.title2.weight(.semibold))

                VStack(spacing: 12) {
                    ForEach(MathOperation.allCases) { operation in
                        Toggle(operation.displayName, isOn: Binding(
                            get: { config.selectedOperations.contains(operation) },
                            set: { isOn in
                                if isOn {
                                    config.selectedOperations.append(operation)
                                } else {
                                    config.selectedOperations.removeAll { $0 == operation }
                                }
                            }
                        ))
                        .toggleStyle(.button)
                        .frame(maxWidth: .infinity)
                    }
                }
                .padding(.horizontal)

                Button("Start") {
                    guard !config.selectedOperations.isEmpty else { return }
                    onStart(config)
                }
                .buttonStyle(.borderedProminent)
                .disabled(config.selectedOperations.isEmpty)
                .font(.title3.weight(.semibold))
                .padding(.horizontal, 40)
            }
            .padding()
            .navigationTitle("Math Quiz")
        }
    }
    ```

    ```swift
    import XCTest

    final class NumberPadUITests: XCTestCase {
        var app: XCUIApplication!

        override func setUpWithError() throws {
            continueAfterFailure = false
            app = XCUIApplication()
            app.launch()
            // Navigate to GameView: tap "Addition" toggle to ensure it's selected, then Start
            let additionButton = app.buttons["Addition"]
            if additionButton.exists && !additionButton.isSelected {
                additionButton.tap()
            }
            app.buttons["Start"].tap()
        }

        func testPadButtonCount() throws {
            // After navigation, number pad should be visible with 12 buttons
            // Buttons identified by accessibility labels: 1-9, Delete, 0, Submit
            let padButtons = ["1","2","3","4","5","6","7","8","9","Delete","0","Submit"]
            for label in padButtons {
                XCTAssertTrue(app.buttons[label].exists, "Button '\(label)' not found")
            }
        }

        func testPadButtonMinimumSize() throws {
            // Verify each button is at least 70pt tall
            let buttonLabels = ["1","2","3","4","5","6","7","8","9","Delete","0","Submit"]
            for label in buttonLabels {
                let button = app.buttons[label]
                XCTAssertGreaterThanOrEqual(button.frame.height, 70,
                    "Button '\(label)' height \(button.frame.height) is less than 70pt")
            }
        }
    }
    ```
