    ```swift
    import Foundation
    import Observation

    @Observable
    final class GameViewModel {
        // MARK: - Published state
        private(set) var currentQuestion: MathQuestion?
        private(set) var answerBuffer: String = ""
        private(set) var score: Int = 0
        private(set) var elapsedSeconds: Int = 0
        private(set) var feedbackState: FeedbackState = .idle
        private(set) var sessionActive: Bool = false

        // MARK: - Private state
        private var config: GameConfig = .default
        private var sessionStartDate: Date = .now
        private var questionStartDate: Date = .now
        private var backgroundEntryDate: Date?

        // MARK: - Dependencies
        private let questionGenerator: QuestionGenerator
        private let soundPlayer: any SoundPlayerProtocol

        init(questionGenerator: QuestionGenerator = QuestionGenerator(),
             soundPlayer: any SoundPlayerProtocol = SoundPlayer()) {
            self.questionGenerator = questionGenerator
            self.soundPlayer = soundPlayer
        }

        // MARK: - Session lifecycle
        func startSession(config: GameConfig) {
            self.config = config
            score = 0
            elapsedSeconds = 0
            answerBuffer = ""
            feedbackState = .idle
            sessionStartDate = .now
            sessionActive = true
            nextQuestion()
        }

        func pauseSession() {
            backgroundEntryDate = .now
        }

        func resumeSession() {
            guard let bg = backgroundEntryDate else { return }
            let duration = Date.now.timeIntervalSince(bg)
            sessionStartDate = sessionStartDate.addingTimeInterval(duration)
            backgroundEntryDate = nil
        }

        func tick() {
            guard sessionActive else { return }
            elapsedSeconds = Int(Date.now.timeIntervalSince(sessionStartDate))
        }

        // MARK: - Input
        func appendDigit(_ digit: String) {
            guard feedbackState == .idle else { return }
            guard answerBuffer.count < 5 else { return }   // reasonable max
            answerBuffer += digit
        }

        func deleteLastDigit() {
            guard !answerBuffer.isEmpty else { return }
            answerBuffer.removeLast()
        }

        func submitAnswer() {
            guard feedbackState == .idle,
                  let question = currentQuestion,
                  let answer = Int(answerBuffer) else { return }

            let elapsed = Date.now.timeIntervalSince(questionStartDate)

            if answer == question.correctAnswer {
                let base = 10
                let bonus = max(0, 5 - Int(elapsed))
                score += base + bonus
                soundPlayer.playCorrect()
                feedbackState = .correct
            } else {
                soundPlayer.playIncorrect()
                feedbackState = .incorrect
            }

            answerBuffer = ""
            Task { @MainActor in
                try? await Task.sleep(for: .milliseconds(600))
                feedbackState = .idle
                if self.sessionActive { self.nextQuestion() }
            }
        }

        // MARK: - Private helpers
        private func nextQuestion() {
            currentQuestion = questionGenerator.generate(config: config)
            questionStartDate = .now
        }
    }
    ```
