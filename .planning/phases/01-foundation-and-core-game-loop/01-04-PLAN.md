    Step 2 — Create `Math Quiz/Services/SoundPlayer.swift`:
    ```swift
    import AVFoundation
    import Foundation

    final class SoundPlayer: SoundPlayerProtocol {
        private var correctPlayer: AVAudioPlayer?
        private var incorrectPlayer: AVAudioPlayer?

        init() {
            correctPlayer   = loadPlayer(named: "correct",   ext: "mp3")
            incorrectPlayer = loadPlayer(named: "incorrect", ext: "mp3")
            correctPlayer?.prepareToPlay()
            incorrectPlayer?.prepareToPlay()
        }

        func playCorrect() {
            correctPlayer?.currentTime = 0
            correctPlayer?.play()
        }

        func playIncorrect() {
            incorrectPlayer?.currentTime = 0
            incorrectPlayer?.play()
        }

        private func loadPlayer(named name: String, ext: String) -> AVAudioPlayer? {
            guard let url = Bundle.main.url(forResource: name, withExtension: ext) else {
                // Audio file not found — log in debug, return nil (silent fallback)
                return nil
            }
            return try? AVAudioPlayer(contentsOf: url)
        }
    }
    ```

    The `nil` guard in `loadPlayer` means missing audio files result in silent operation (no crash) — SoundPlayerTests will pass even in the Simulator where AVAudioPlayer cannot play to hardware.
